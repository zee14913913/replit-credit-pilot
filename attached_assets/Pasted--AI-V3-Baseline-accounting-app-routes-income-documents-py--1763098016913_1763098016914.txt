from fastapi import APIRouter, UploadFile, File, Form, Depends, HTTPException
from sqlalchemy.orm import Session
from accounting_app.db import get_db
from accounting_app.services.file_storage_manager import FileStorageManager
from accounting_app.services.unified_file_service import UnifiedFileService
from accounting_app.services.pdf_parser import PDFParser
from datetime import datetime
import os

router = APIRouter(prefix="/api/income-documents", tags=["Income Documents"])

@router.post("/upload")
async def upload_income_document(
    customer_id: int = Form(...),
    document_type: str = Form(...),   # salary_slip / tax_return / epf / bank_inflow
    document_month: str = Form(...),  # YYYY-MM
    file: UploadFile = File(...),
    db: Session = Depends(get_db)
):

    # 1. 保存文件到 storage/income/
    saved_path = FileStorageManager.save_upload(
        file=file,
        module="income",
        subfolder=document_type
    )

    if not saved_path:
        raise HTTPException(status_code=500, detail="无法保存文件")

    # 2. 创建 RawDocument（原件封存）
    raw_doc = UnifiedFileService.create_raw_document(
        db=db,
        filename=file.filename,
        file_path=saved_path,
        module="income",
        content_type=file.content_type
    )

    # 3. OCR / PDF 解析，提取收入金额
    parser = PDFParser()
    parsed = parser.parse_pdf(saved_path)

    extracted_income = parsed.get("extracted_income", 0)
    confidence = parsed.get("confidence", 0.0)
    raw_text = parsed.get("raw_text", "")

    # 4. 写入 file_index
    file_index = UnifiedFileService.register_file(
        db=db,
        company_id=1,
        filename=file.filename,
        file_path=saved_path,
        module="income",
        file_category="income_proof",
        file_type=document_type,
        period=document_month,
        related_entity_type="customer_id",
        related_entity_id=customer_id,
        raw_document_id=raw_doc.id
    )

    # 5. 返回 OCR 结果
    return {
        "status": "success",
        "customer_id": customer_id,
        "document_type": document_type,
        "document_month": document_month,
        "file_id": file_index.id,
        "raw_document_id": raw_doc.id,
        "income_detected": extracted_income,
        "confidence": confidence,
        "raw_text": raw_text,
    }
